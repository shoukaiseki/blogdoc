黹_,不^建h多手绦,@舆@些命令就成T成本能
#!/bin/sh
#dbstart.sh
lsnrctl start
echo "lsnrctlO服粘晒_!"
isqlplusctl start
echo "isqlplusctl服粘晒_!"
sqlplus /nolog <<EOF
connect / as sysdba
startup;
quit
EOF
echo "斐晒_!"

#!/bin/sh
#dbshut.sh
lsnrctl stop
echo "lsnrctlO成功P]"
isqlplusctl stop
echo "isqlplusctl服粘晒P]"
$SQLDBA <<EOF
connect / as sysdba
shutdown immediate
quit
EOF
echo "斐晒P]"
